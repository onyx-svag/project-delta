-- Получение корневой части персонажа
local function getRootPart()
    local character = localPlayer.Character
    if not character then return nil end
    return character:FindFirstChild("HumanoidRootPart")
end

-- Симуляция клика по элементу интерфейса
local function simulateClick(button)
    if not button then return end
    local x = button.AbsolutePosition.X + button.AbsoluteSize.X / 2
    local y = button.AbsolutePosition.Y + button.AbsoluteSize.Y / 2
    VirtualInputManager:SendMouseButtonEvent(x, y, 0, true, game, 0)
    task.wait(0.05)
    VirtualInputManager:SendMouseButtonEvent(x, y, 0, false, game, 0)
end

-- Перемещение персонажа к заданной позиции (Fly-мод)
local function teleportTo(targetPosition, speed)
    speed = speed or state.flySpeed
    while state.running do
        local root = getRootPart()
        if not root then
            task.wait(0.1)
        else
            local direction = targetPosition - root.Position
            local distance = direction.Magnitude
            if distance < 5 then
                return
            else
                local step = direction.Unit * math.min(speed * 0.05, distance)
                root.CFrame = root.CFrame + step
                task.wait(0.05)
            end
        end
    end
end

-- Поиск ближайшей цели (врага/объекта) по списку редкостей
local function findNearestTarget()
    local root = getRootPart()
    if not root then return nil, 0 end

    local nearest = nil
    local minDistance = math.huge

    for _, descendant in ipairs(workspace:GetDescendants()) do
        if descendant:IsA("Model") then
            for _, rarity in ipairs(rarities) do
                if descendant.Name == rarity then
                    local targetPart = descendant:FindFirstChild("HitboxPart") 
                        or descendant:FindFirstChild("Root") 
                        or descendant:FindFirstChild("RootPart")

                    if targetPart then
                        local distance = (root.Position - targetPart.Position).Magnitude
                        if distance < minDistance then
                            minDistance = distance
                            nearest = targetPart
                        end
                    end
                end
            end
        end
    end
    return nearest, minDistance
end

-- Ожидание разморозки персонажа
local function waitUnfreeze()
    local startTime = tick()
    while tick() - startTime < 10 do
        if not state.running then return end
        local root = getRootPart()
        if root then
            if root.Anchored then
                -- Если персонаж заморожен, ждем разморозки
                local freezeStart = tick()
                while tick() - freezeStart < 45 do
                    if not state.running then return end
                    local currentRoot = getRootPart()
                    if not currentRoot or not currentRoot.Anchored then
                        task.wait(0.8)
                        return
                    end
                    task.wait(0.1)
                end
            else
                task.wait(0.1)
            end
        end
    end
end

-- Главный цикл автофарма (MAIN AUTOFARM)
local function startAutofarm()
    updateStatus("STARTING IN 3s...", colors.SubText)
    task.wait(3)

    while state.running do
        state.roundCount = state.roundCount + 1
        roundsLabel.Text = tostring(state.roundCount)
        
        updateStatus("ROUND " .. state.roundCount .. " | MOVING", colors.Accent)
        teleportTo(POSITION_KICK, 550)

        if not state.running then
            updateStatus("STOPPED", colors.Red)
            return
        end

        task.wait(0.3)
        
        -- Поиск кнопки пинка ("KickButton")
        local playerGui = localPlayer.PlayerGui
        local hud = playerGui:FindFirstChild("HUD")
        local kickBtn = nil

        if hud then
            kickBtn = hud:FindFirstChild("KickButton")
        end

        if not kickBtn then
            -- Глубокий поиск, если кнопка не найдена по стандартному пути
            for _, desc in ipairs(playerGui:GetDescendants()) do
                if desc.Name == "KickButton" and desc:IsA("ImageButton") then
                    kickBtn = desc
                    break
                end
            end
        end

        if not kickBtn then
            updateStatus("KICK BUTTON NOT FOUND!", colors.Red)
            task.wait(1)
        else
            updateStatus("MOVING MOUSE & CLICKING...", colors.Accent)
            
            -- Вычисляем абсолютные координаты клика
            local inset = GuiService:GetGuiInset()
            local targetX = kickBtn.AbsolutePosition.X + kickBtn.AbsoluteSize.X / 2 + inset.X
            local targetY = kickBtn.AbsolutePosition.Y + kickBtn.AbsoluteSize.Y / 2 + inset.Y

            -- Эмулируем движение мыши и клик
            if mousemoveabs then
                mousemoveabs(targetX, targetY)
                task.wait(0.05)
                VirtualInputManager:SendMouseButtonEvent(targetX, targetY, 0, true, game, 0)
                task.wait(0.05)
                VirtualInputManager:SendMouseButtonEvent(targetX, targetY, 0, false, game, 0)
                task.wait(0.1)

                -- Логика решения мини-игры ("KickMinigame")
                updateStatus("MINIGAME...", colors.Accent)
                local movingBar = nil

                for i = 1, 50 do
                    if not state.running then break end
                    local minigameGui = playerGui:FindFirstChild("KickMinigame")
                    if minigameGui and minigameGui.Enabled then
                        for _, desc in pairs(minigameGui:GetDescendants()) do
                            if desc.Name == "MovingBar" and desc.Visible then
                                movingBar = desc
                                break
                            end
                        end
                    end
                    if movingBar then break end
                    task.wait(0.1)
                end

                if movingBar then
                    local timeout = tick() + 6
                    local lastScale = 0

                    while tick() < timeout and state.running and movingBar.Parent do
                        local success, currentScale = pcall(function()
                            return movingBar.Size.Y.Scale
                        end)

                        if success and currentScale then
                            -- Пытаемся поймать идеальный тайминг (Sweet Spot >= 0.93)
                            if currentScale > lastScale then
                                lastScale = currentScale
                                if currentScale >= 0.93 then
                                    local viewport = workspace.CurrentCamera.ViewportSize
                                    mousemoveabs(viewport.X / 2, viewport.Y / 2)
                                    task.wait(0.02)
                                    VirtualInputManager:SendMouseButtonEvent(viewport.X / 2, viewport.Y / 2, 0, true, game, 0)
                                    task.wait(0.05)
                                    VirtualInputManager:SendMouseButtonEvent(viewport.X / 2, viewport.Y / 2, 0, false, game, 0)
                                    updateStatus("PERFECT! " .. math.floor(currentScale * 100) .. "%", colors.Green)
                                    break
                                end
                            else
                                -- Если шкала пошла на спад
                                if lastScale >= 0.93 then
                                    local viewport = workspace.CurrentCamera.ViewportSize
                                    mousemoveabs(viewport.X / 2, viewport.Y / 2)
                                    task.wait(0.02)
                                    VirtualInputManager:SendMouseButtonEvent(viewport.X / 2, viewport.Y / 2, 0, true, game, 0)
                                    task.wait(0.05)
                                    VirtualInputManager:SendMouseButtonEvent(viewport.X / 2, viewport.Y / 2, 0, false, game, 0)
                                    updateStatus("PERFECT! " .. math.floor(lastScale * 100) .. "%", colors.Green)
                                    break
                                end
                                lastScale = currentScale
                            end
                        end
                        task.wait(0.016)
                    end
                end

                -- Ожидание завершения анимаций и разморозки
                updateStatus("WAITING UNFREEZE...", colors.SubText)
                waitUnfreeze()

                if state.running then
                    -- Возвращаемся на позицию ожидания
                    updateStatus("RETURNING TO BASE", colors.Accent)
                    teleportTo(POSITION_WAIT, state.flySpeed)

                    if state.running then
                        -- Сканируем окружение на появление новой волны
                        updateStatus("WAITING WAVE...", colors.SubText)
                        task.wait(5)
                        local waveTimeout = tick() + 120
                        local waveDetected = false

                        while tick() < waveTimeout and state.running and not waveDetected do
                            local target, distance = findNearestTarget()
                            if target and distance <= 250 then
                                waveDetected = true
                                task.wait(0.2)
                            end
                            task.wait(0.1)
                        end

                        if waveDetected then
                            updateStatus("WAVE! SAFE POS", colors.Green)
                            teleportTo(POSITION_SAFE, 450)
                            task.wait(1)
                        else
                            updateStatus("WAVE TIMEOUT", colors.Red)
                        end
                    end
                end
            else
                -- Альтернативный метод перемещения мыши через относительные координаты
                if mousemoverel then
                    local mouse = localPlayer:GetMouse()
                    mousemoverel(targetX - mouse.X, targetY - mouse.Y)
                end
            end
        end
    end
end

-- Автокликер бонусов (AUTO BONUS CLICKER)
local function handleBonusActivation(bonusBtn)
    if not bonusBtn or not bonusBtn.Visible then return end
    
    local success = pcall(function()
        bonusBtn:Activate()
    end)
    
    if not success then
        simulateClick(bonusBtn)
    end
    
    bonusStatusLabel.Text = "Clicked at " .. os.date("%H:%M:%S")
    bonusStatusLabel.TextColor3 = colors.Accent
end

local function startBonusLoop()
    bonusStatusLabel.Text = "Active, listening for button..."
    bonusStatusLabel.TextColor3 = colors.SubText

    -- Подключение к динамически появляющимся кнопкам
    local connection = localPlayer.PlayerGui.DescendantAdded:Connect(function(descendant)
        if state.bonusRunning and descendant.Name == "Bonus" then
            if descendant:IsA("TextButton") or descendant:IsA("ImageButton") then
                task.wait(0.15)
                handleBonusActivation(descendant)
                
                -- Отслеживаем видимость
                descendant:GetPropertyChangedSignal("Visible"):Connect(function()
                    if descendant.Visible and state.bonusRunning then
                        task.wait(0.1)
                        handleBonusActivation(descendant)
                    end
                end)
            end
        end
    end)

    -- Постоянная проверка статических кнопок
    while state.bonusRunning do
        local upgrades = localPlayer.PlayerGui:FindFirstChild("KickUpgrades")
        if upgrades then
            local bonus = upgrades:FindFirstChild("Bonus")
            if bonus and bonus:IsA("GuiObject") and bonus.Visible then
                handleBonusActivation(bonus)
                task.wait(0.5)
            end
        end
        task.wait(0.1)
    end

    connection:Disconnect()
    bonusStatusLabel.Text = "Disabled"
    bonusStatusLabel.TextColor3 = colors.SubText
end
