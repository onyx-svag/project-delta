local function registerToggle(name, defaultValue)
    local item = { Value = defaultValue or false }
    item.Set = function(self, val)
        self.Value = val
        if self._windElem then
            self._windElem:Set(val)
        end
    end
    toggles[name] = item
    return item
end

local function registerOption(name, defaultValue)
    local item = { Value = defaultValue }
    options[name] = item
    return item
end

-- Функция очистки эффектов при выгрузке меню
local function cleanEffectsAndCheat()
    pcall(function()
        local zombiesLocal = workspace:FindFirstChild("Zombies_Local")
        if zombiesLocal then
            for _, zombie in ipairs(zombiesLocal:GetChildren()) do
                local hl = zombie:FindFirstChildOfClass("Highlight")
                if hl then hl:Destroy() end
            end
        end
    end)
    pcall(function()
        local char = localPlayer.Character
        if char then
            local root = char:FindFirstChild("HumanoidRootPart")
            if root then
                local bv = root:FindFirstChildOfClass("BodyVelocity")
                if bv then bv:Destroy() end
            end
        end
    end)
    pcall(function()
        local chair = workspace:FindFirstChild("invischair")
        if chair then chair:Destroy() end
        local char = localPlayer.Character
        if char then
            for _, part in pairs(char:GetDescendants()) do
                if part:IsA("BasePart") and part.Name ~= "HumanoidRootPart" then
                    part.Transparency = 0
                end
            end
        end
    end)
    pcall(function()
        local char = localPlayer.Character
        if char then
            for _, part in ipairs(char:GetDescendants()) do
                if part:IsA("BasePart") then
                    part.CanCollide = true
                end
            end
            local hum = char:FindFirstChildOfClass("Humanoid")
            if hum then
                hum.WalkSpeed = 16
                hum.JumpHeight = 7.2
            end
        end
    end)
    pcall(function()
        Lighting.Brightness = originalLighting.Brightness
        Lighting.Ambient = originalLighting.Ambient
        Lighting.OutdoorAmbient = originalLighting.OutdoorAmbient
        Lighting.FogEnd = originalLighting.FogEnd
        Lighting.ClockTime = originalLighting.ClockTime
    end)
end

-- Расчет текущего значения кадров в секунду (FPS)
local fpsBuffer = {}
RunService.Heartbeat:Connect(function(dt)
    table.insert(fpsBuffer, 1 / dt)
    if #fpsBuffer > 30 then
        table.remove(fpsBuffer, 1)
    end
end)

local function getAverageFPS()
    if #fpsBuffer == 0 then return 0 end
    local total = 0
    for _, fps in ipairs(fpsBuffer) do
        total = total + fps
    end
    return math.floor(total / #fpsBuffer)
end

        if val then
            task.spawn(function()
                local targetColor = Color3.fromRGB(170, 0, 255)
                while toggles.AutoCollectShards.Value do
                    pcall(function()
                        local char = localPlayer.Character
                        local root = char and char:FindFirstChild("HumanoidRootPart")
                        if root then
                            for _, part in ipairs(workspace:GetDescendants()) do
                                if not toggles.AutoCollectShards.Value then return end
                                if part:IsA("BasePart") and not part.Anchored then
                                    local colorMatch = math.abs(part.Color.R - targetColor.R) + math.abs(part.Color.G - targetColor.G) + math.abs(part.Color.B - targetColor.B) < 0.1
                                    if colorMatch then
                                        local dist = (root.Position - part.Position).Magnitude
                                        if dist < 80 then
                                            root.CFrame = CFrame.new(part.Position + Vector3.new(0, 3, 0))
                                            task.wait(0.05)
                                        end
                                    end
                                end
                            end
                        end
                    end)
                    task.wait(1)
                end
            end)
        end
    end
})

local queueRemotes = ReplicatedStorage:FindFirstChild("QueueRemotes")
local queueSettings = {
    difficulty = "Normal",
    partySize = 1,
    map = "RooftopSiege",
    autoOn = false,
    hookConn = nil,
    updateConn = nil
}

local function teleportToClosestQueue()
    local queues = workspace:FindFirstChild("Queues")
    local char = localPlayer.Character
    local root = char and char:FindFirstChild("HumanoidRootPart")
    if queues and root then
        local closest = nil
        local minDist = math.huge
        for _, q in ipairs(queues:GetChildren()) do
            local pad = q:FindFirstChild("BillboardPart") or q:FindFirstChildWhichIsA("BasePart")
            if pad then
                local d = (root.Position - pad.Position).Magnitude
                if d < minDist then
                    minDist = d
                    closest = pad
                end
            end
        end
        if closest then
            root.CFrame = CFrame.new(closest.Position + Vector3.new(0, 3, 0))
        end
    end
end

local function requestQueueJoin()
    if queueRemotes and queueRemotes:FindFirstChild("CreateParty") then
        pcall(function()
            queueRemotes.CreateParty:FireServer(queueSettings.partySize, queueSettings.difficulty, queueSettings.map)
        end)
    end
end

tabQueue:Toggle({
    Title = "Full Auto Queue",
    Value = false,
    Callback = function(val)
        queueSettings.autoOn = val
        if val then
            if queueRemotes then
                local showUI = queueRemotes:FindFirstChild("ShowPartySizeUI")
                if showUI then
                    if queueSettings.hookConn then queueSettings.hookConn:Disconnect() end
                    queueSettings.hookConn = showUI.OnClientEvent:Connect(function(eventState)
                        if not eventState or not queueSettings.autoOn then return end
                        task.wait(0.1)
                        requestQueueJoin()
                    end)
                end
                local updateQ = queueRemotes:FindFirstChild("QueueUpdate")
                if updateQ then
                    if queueSettings.updateConn then queueSettings.updateConn:Disconnect() end
                    queueSettings.updateConn = updateQ.OnClientEvent:Connect(function(updateData)
                        if not queueSettings.autoOn then return end
                        local action = type(updateData) == "table" and updateData.Action
                        if action == "Left" or action == "TeleportFailed" then
                            task.wait(3)
                            teleportToClosestQueue()
                        end
                    end)
                end
            end
            teleportToClosestQueue()
        else
            if queueSettings.hookConn then queueSettings.hookConn:Disconnect(); queueSettings.hookConn = nil end
            if queueSettings.updateConn then queueSettings.updateConn:Disconnect(); queueSettings.updateConn = nil end
            pcall(function()
                if queueRemotes and queueRemotes:FindFirstChild("LeaveQueue") then
                    queueRemotes.LeaveQueue:FireServer()
                end
            end)
        end
    end
})
