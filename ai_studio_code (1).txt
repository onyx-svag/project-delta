local function getHRP()
    local char = localPlayer.Character
    return char and char:FindFirstChild("HumanoidRootPart")
end

-- Клик по GUI объекту через посылку клика мышкой по координатам
local function clickGuiObject(guiObject)
    if not guiObject then return end
    local posX = guiObject.AbsolutePosition.X + guiObject.AbsoluteSize.X / 2
    local posY = guiObject.AbsolutePosition.Y + guiObject.AbsoluteSize.Y / 2
    VirtualInputManager:SendMouseButtonEvent(posX, posY, 0, true, game, 0)
    task.wait(0.05)
    VirtualInputManager:SendMouseButtonEvent(posX, posY, 0, false, game, 0)
end

-- Полет / Перемещение к точке
local function flyTo(targetPos, speed)
    speed = speed or state.flySpeed
    while state.running do
        local hrp = getHRP()
        if not hrp then
            task.wait(0.1)
        else
            local direction = targetPos - hrp.Position
            local distance = direction.Magnitude
            if distance < 5 then
                return
            else
                hrp.CFrame = hrp.CFrame + direction.Unit * math.min(speed * 0.05, distance)
                task.wait(0.05)
            end
        end
    end
end

-- Поиск ближайшей цели (из списка редкостей)
local function findNearestTarget()
    local hrp = getHRP()
    if not hrp then return nil, 0 end

    local nearest, minDistance = nil, math.huge
    for _, desc in ipairs(workspace:GetDescendants()) do
        if desc:IsA("Model") then
            for _, rarity in ipairs(RARITIES) do
                if desc.Name == rarity then
                    local targetPart = desc:FindFirstChild("HitboxPart") or desc:FindFirstChild("Root") or desc:FindFirstChild("RootPart")
                    if targetPart then
                        local distance = (hrp.Position - targetPart.Position).Magnitude
                        if distance < minDistance then
                            minDistance = distance
                            nearest = targetPart
                        end
                    end
                end
            end
        end
    end
    return nearest, minDistance
end

-- Основной цикл фарма
local function mainLoop()
    setStatus("STARTING IN 3s...", COLORS.Accent)
    task.wait(3)

    while state.running do
        state.roundCount = state.roundCount + 1
        roundsLabel.Text = tostring(state.roundCount)
        setStatus("ROUND " .. tostring(state.roundCount), COLORS.Accent)

        -- Летим на позицию удара
        flyTo(KICK_POS, 550)
        if not state.running then
            setStatus("STOPPED", COLORS.Red)
            return
        end
        task.wait(0.3)

        -- Проверяем наличие мини-игры в PlayerGui
        local playerGui = localPlayer:WaitForChild("PlayerGui")
        local minigameUi = playerGui:FindFirstChild("Minigame") -- Имя изменено для читаемости
        if minigameUi then
            local kickButton = minigameUi:FindFirstChild("KickButton")
            if not kickButton then
                -- Поиск кнопки пинка через дескенданты
                for _, child in ipairs(minigameUi:GetDescendants()) do
                    if child.Name == "KickButton" and child:IsA("GuiButton") then
                        kickButton = child
                        break
                    end
                end
            end

            if not kickButton then
                setStatus("KICK BUTTON NOT FOUND!", COLORS.Red)
                task.wait(1)
            else
                setStatus("MOVING MOUSE & CLICKING...", COLORS.Accent)
                -- Расчет позиции клика по кнопке
                local clickX = kickButton.AbsolutePosition.X + kickButton.AbsoluteSize.X / 2
                local clickY = kickButton.AbsolutePosition.Y + kickButton.AbsoluteSize.Y / 2
                local inset = GuiService:GetGuiInset()
                local targetX = clickX + inset.X
                local targetY = clickY + inset.Y

                -- Эмуляция перемещения мыши и клика
                if mousemoveabs then
                    mousemoveabs(targetX, targetY)
                    task.wait(0.05)
                    VirtualInputManager:SendMouseButtonEvent(targetX, targetY, 0, true, game, 0)
                    task.wait(0.05)
                    VirtualInputManager:SendMouseButtonEvent(targetX, targetY, 0, false, game, 0)
                    task.wait(0.1)

                    setStatus("MINIGAME...", COLORS.Accent)
                    -- Поиск полосы мини-игры
                    local movingBar = nil
                    for _, d in ipairs(minigameScreen:GetDescendants()) do
                        if d.Name == "MovingBar" and d.Visible then
                            movingBar = d
                            break
                        end
                    end

                    if movingBar then
                        local perfectHitTime = tick() + 6
                        while movingBar and movingBar.Parent and state.running and tick() < timeout do
                            local currentScale = movingBar.Size.Y.Scale
                            if currentScale >= 0.93 then
                                local cameraSize = workspace.CurrentCamera.ViewportSize
                                mousemoveabs(cameraSize.X / 2, cameraSize.Y / 2)
                                task.wait(0.02)
                                VirtualInputManager:SendMouseButtonEvent(camera.X / 2, camera.Y / 2, 0, true, game, 0)
                                task.wait(0.05)
                                VirtualInputManager:SendMouseButtonEvent(camera.X / 2, camera.Y / 2, 0, false, game, 0)

                                setStatus("PERFECT! " .. math.floor(currentScale * 100) .. "%", COLORS.Green)
                                break
                            end
                            task.wait(0.016)
                        end
                    end
                end
            end
        end

        task.wait(0.5)
        setStatus("WAITING UNFREEZE...", COLORS.SubText)
        
        -- Возвращаемся на базу
        setStatus("RETURNING TO BASE", COLORS.Accent)
        flyTo(WAIT_POS, state.flySpeed)

        -- Ожидаем начала новой волны
        setStatus("WAITING WAVE...", COLORS.SubText)
        task.wait(5)
        local waveFound = false
        local waveTimeout = tick() + 120

        while not waveDetected and tick() < waveTimeout and state.running do
            local nearestObj, distance = findNearestTarget()
            if nearestObj and distance <= 250 then
                waveDetected = true
                setStatus("WAVE! SAFE POS", COLORS.Green)
                flyTo(SAFE_POS, 450)
                task.wait(1)
            end
            task.wait(0.2)
        end
    end
    setStatus("READY", COLORS.Accent)
end

-- Автоматический клик по кнопке Bonus
local function startBonusLoop()
    bonusStatusText.Text = "Active, listening for button..."
    bonusStatusText.TextColor3 = COLORS.SubText

    local connection
    connection = localPlayer.PlayerGui.DescendantAdded:Connect(function(descendant)
        if not state.bonusRunning then return end
        if descendant.Name == "Bonus" and (descendant:IsA("TextButton") or descendant:IsA("ImageButton")) then
            task.wait(0.15)
            simulateClick(descendant)
        end
    end)

    while state.bonusRunning do
        local bonusBtn = localPlayer.PlayerGui:FindFirstChild("Bonus", true)
        if bonusBtn and bonusBtn:IsA("GuiObject") and bonusBtn.Visible then
            simulateClick(bonusBtn)
            task.wait(0.5)
        end
        task.wait(1)
    end

    connection:Disconnect()
    bonusStatusText.Text = "Disabled"
    bonusStatusText.TextColor3 = COLORS.SubText
end

-- Анти-AFK
local function startAntiAfkLoop()
    afkStatusText.Text = "Active (Looping 2 min)"
    afkStatusText.TextColor3 = COLORS.Accent

    while state.antiAfkRunning do
        for i = 1, 120 do
            if not state.antiAfkRunning then break end
            task.wait(1)
        end

        if state.antiAfkRunning then
            afkStatusText.Text = "Jumping! " .. os.date("%H:%M:%S")
            
            -- Убираем предмет (Кнопка '1')
            VirtualInputManager:SendKeyEvent(true, Enum.KeyCode.One, false, game)
            task.wait(0.15)
            VirtualInputManager:SendKeyEvent(false, Enum.KeyCode.One, false, game)
            task.wait(1.5)

            -- Прыгаем (Кнопка 'Space')
            VirtualInputManager:SendKeyEvent(true, Enum.KeyCode.Space, false, game)
            task.wait(0.1)
            VirtualInputManager:SendKeyEvent(false, Enum.KeyCode.Space, false, game)
            task.wait(0.5)

            -- Достаем обратно
            VirtualInputManager:SendKeyEvent(true, Enum.KeyCode.One, false, game)
            task.wait(0.15)
            VirtualInputManager:SendKeyEvent(false, Enum.KeyCode.One, false, game)
            
            afkStatusText.Text = "Active (Looping 2 min)"
            afkStatusText.TextColor3 = COLORS.Accent
        end
    end
    afkStatusText.Text = "Disabled"
    afkStatusText.TextColor3 = COLORS.SubText
end
